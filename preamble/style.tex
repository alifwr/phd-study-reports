% --- capture the current topic (\part) name for use in the header ---
% report.cls keeps no part mark, so patch \part to remember its title.
\newcommand{\topicname}{}
\let\origpart\part
\renewcommand{\part}[1]{\renewcommand{\topicname}{#1}\origpart{#1}}

% --- headers / footers ---
% Header: current topic (left) | current chapter (right)
% Footer: report title (left) | page (center) | author (right)
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\small\itshape\topicname}
\fancyhead[R]{\small\itshape\nouppercase{\leftmark}}
\fancyfoot[L]{\footnotesize\reporttitle}
\fancyfoot[C]{\footnotesize\thepage}
\fancyfoot[R]{\footnotesize\reportauthor}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0.4pt}
% Chapter-opening / front-matter pages: keep the same footer, no header rule.
\fancypagestyle{plain}{%
  \fancyhf{}%
  \fancyfoot[L]{\footnotesize\reporttitle}%
  \fancyfoot[C]{\footnotesize\thepage}%
  \fancyfoot[R]{\footnotesize\reportauthor}%
  \renewcommand{\headrulewidth}{0pt}%
  \renewcommand{\footrulewidth}{0.4pt}%
}

% --- code listing default style ---

\definecolor{codecomment}{rgb}{0.0,0.5,0.0}
\definecolor{codekeyword}{rgb}{0.0,0.0,0.6}
\lstdefinestyle{default}{
  backgroundcolor=\color{codegray},
  commentstyle=\color{codecomment},
  keywordstyle=\color{codekeyword}\bfseries,
  basicstyle=\ttfamily\footnotesize,
  breaklines=true,
  captionpos=b,
  frame=single,
  numbers=left,
  numberstyle=\tiny,
  showstringspaces=false,
  tabsize=2,
}
